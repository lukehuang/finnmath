package com.github.ltennstedt.finnmath.kotlin.linear

import com.github.ltennstedt.finnmath.core.linear.BigDecimalMatrix

operator fun BigDecimalMatrix.plus(summand: BigDecimalMatrix) = add(summand)

operator fun BigDecimalMatrix.minus(subtrahend: BigDecimalMatrix) = subtract(subtrahend)

operator fun BigDecimalMatrix.times(factor: BigDecimalMatrix) = multiply(factor)

operator fun BigDecimalMatrix.unaryMinus() = negate()
